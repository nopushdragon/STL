class myArray {
public:
	size_t size() const {	// size_t 매우 큰 값까쥐 가능, const 읽기전용
		return N;
	}

	int& operator[](int idx) {	// [] 연산자 오버로딩, idx는 배열의 인덱스
		return arr[idx];

	T* begin(){	// begin() 멤버 함수, 배열의 시작 주소 반환
		return &arr[0];
	}

	T* end() {	// end() 멤버 함수, 배열의 끝 주소 반환
		return &arr[N];
	}

private:
	T arr[N];
