	//시간 정보를 얻는다
	//std::chrono::system_clock::time_point now = std::chrono::system_clock::now();
	auto now = std::chrono::system_clock::now();	// time epoch(시간의 기원, 1970년 1월1일 자정)로부터 1초마다 증가한 tick 수

	using namespace std::chrono_literals;

	out << "\n" << "\n";
	out << "===============================================" << std::endl;
	out << "저장한 시간 -" << now + 9h << std::endl;	// 이것을 2026년03월00일 월요일 오후 2:34 << 이런 식으로 바꿔보기
	out << "===============================================" << std::endl;
	out << "\n";
	
	// std::copy(어디에서, 어디까지를, 어디로);
	std::copy(std::istreambuf_iterator<char>{in}, {},
		std::ostream_iterator<char>{out});
}

#include <iostream>
#include "save.h"

// 템플릿 - 동일한 내용의 코드를 자료형만 바꿔서 사용할거시야.

// 템플릿 함수는 선언과 정의를 분리할 수 없다. <- 이유 다시 한 번 알아보면 좋을 느낌
template<class T> //실전에서는 이런 식으로 T를 사용하지 않음, 문장형으로 지음.
void change(T& a, T& b) 
{
	std::cout << "템플릿 함수가 호출됨" << std::endl;
	T temp{ a };
	a = b;
	b = temp;
}

// double에 대한 specialization - 템플릿 특수화
template<> 
void change<double>(double& a, double& b)
{
	std::cout << "double에 대한 specialization이 호출됨" << std::endl;
	double temp{ a };
	a = b;
	b = temp;
}

int main()
{
	double a, b;
	change<double>(a, b);	// 뭘 호출해야 하나? - 호출우선순위 결정
	//이런 식으로 <>으로 내가 해 줄 자료형을 명시적으로 해주면 좋음, 근데 안해도 알아서 해주긴 함.

	std::string c{ "Hello" }, d{ "World" };
	change(c,d);
	
	save("메인.cpp");
}
