//여기 설명부터 다시
//fname부터 하자 -> naming convention

void save(const std::string& fname) 
{
	std::ifstream in(fname);	//RAII 방식 찾아보고 공부하기
	if (not in) {
		std::cout << fname << " - 열 수 없음" << std::endl;
		exit(20260309);
	}


	std::cout << "\"" << fname << "\"" << "를 저장합니다." << std::endl;

	std::ofstream out("2026년 1학기 STL.txt", std::ios::app);

	//시간 정보를 얻는다
	//std::chrono::system_clock::time_point now = std::chrono::system_clock::now();
	auto now = std::chrono::system_clock::now();	// time epoch(시간의 기원, 1970년 1월1일 자정)로부터 1초마다 증가한 tick 수

	using namespace std::chrono_literals;

	out << "\n" << "\n";
	out << "===============================================" << std::endl;
	out << "저장한 시간 -" << now + 9h << std::endl;	// 이것을 2026년03월00일 월요일 오후 2:34 << 이런 식으로 바꿔보기
	out << "===============================================" << std::endl;
	out << "\n";
	
	// std::copy(어디에서, 어디까지를, 어디로);
	std::copy(std::istreambuf_iterator<char>{in}, {},
		std::ostream_iterator<char>{out});
}

void change(int*, int*);

void change(int* a, int* b)
{
	int temp = *a;
	*a = *b;
	*b = temp;
}

{
	// [문제] 의도대로 실행되게 하자
